关于Spring那些事

Spring IOC

  1. 什么是IOC？

    这个大家已经非常熟悉了，控制反转。就是将我们的Bean交给Spring容器来控制。

  2. 如何使用IOC？

    • 自动注入
    // 实体类
@Data
public class Student {
	private int id;
	private String name;
}
    // Repository
@Repository
@Mapper
public class StudentMapper extends BaseMapper<Student>{
	public User findById(int id);
}
    // Service
@Service
public class StudentService {

  // 自动注入
	@Autowired
	StudentMapper stuMapper;
	
	public User findById(int id) {
		return stuMapper.findById(id);
	}
}
    // 主类
@SpringBootApplication
// 扫描注入的包
@MapperScan("com.auto.dao.*")
public class SpringBootMain {

      public static void main(String[] args) {  
       SpringApplication.run(App.class, args);  
      }  
}
    • 手动set方式注入
    // 实体类
@Data
public class Student {
	private int id;
	private String name;
}
    // Repository
@Repository
@Mapper
public class StudentMapper extends BaseMapper<Student>{
	public User findById(int id);
}
    // Service
@Service
public class StudentService {

	StudentMapper stuMapper;
	
	public void setStuMapper(StudentMapper stuMapper) {
		this.stuMapper = stuMapper;
	}
	public User findById(int id) {
		return stuMapper.findById(id);
	}
}
    // config类
@Configuration
public class config {

  // 手动注入Bean
	@Bean
	public StudentMapper mapper() {
		return new StudentMapper();
	}
	
	@Bean
	public StudentService stuService() {
		StudentService s = new StudentService();
		StudentMapper sM = new StudentMapper();
		s.setStuMapper(sM);
	}
}

    以上代码为两种方式的注入方式。
